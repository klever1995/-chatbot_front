import { useState, useEffect, useRef } from 'react'
import { Upload, Edit2, Trash2, DollarSign, MessageSquare, X, Save, File } from 'lucide-react'
import { listarDocumentos, subirDocumento, eliminarDocumento, actualizarPrecio, actualizarMensaje } from '../services/documentos'
import '../styles/Campanas.css'

export default function Campanas() {
  const [documentos, setDocumentos] = useState([])
  const [loading, setLoading] = useState(true)
  const [showModal, setShowModal] = useState(false)
  const [editingDoc, setEditingDoc] = useState(null)
  const [replacingPdf, setReplacingPdf] = useState(null)
  const fileInputRef = useRef(null)
  const [formData, setFormData] = useState({
    archivo: null,
    campania_id: '',
    mensaje_entrega: '',
    precio: ''
  })

  const empresaId = 1 // Por ahora fijo, luego vendrá de configuración

  useEffect(() => {
    loadDocumentos()
  }, [])

  const loadDocumentos = async () => {
    setLoading(true)
    try {
      const data = await listarDocumentos(empresaId)
      setDocumentos(data)
    } catch (error) {
      console.error('Error loading documentos:', error)
    } finally {
      setLoading(false)
    }
  }

  const handleSubir = async () => {
    if (!formData.archivo || !formData.campania_id || !formData.mensaje_entrega || !formData.precio) {
      console.warn('Campos incompletos:', formData);
      return;
    }

    console.log('🟢 handleSubir iniciado con:', {
      archivo: formData.archivo?.name,
      campania_id: formData.campania_id,
      mensaje_entrega: formData.mensaje_entrega,
      precio: formData.precio
    });

    try {
      await subirDocumento(
        empresaId,
        formData.archivo,
        formData.campania_id,
        formData.mensaje_entrega,
        parseFloat(formData.precio)
      );
      console.log('✅ Documento subido exitosamente');
      setShowModal(false);
      setFormData({ archivo: null, campania_id: '', mensaje_entrega: '', precio: '' });
      loadDocumentos();
    } catch (error) {
      console.error('❌ Error en handleSubir:', error);
      alert('Error al subir el documento. Revisa la consola para más detalles.');
    }
  };

  const handleEliminar = async (id) => {
    if (!confirm('¿Estás seguro de eliminar esta campaña?')) return
    try {
      await eliminarDocumento(id)
      loadDocumentos()
    } catch (error) {
      console.error('Error eliminando documento:', error)
    }
  }

  const handleActualizarPrecio = async (id, nuevoPrecio) => {
    try {
      await actualizarPrecio(id, nuevoPrecio)
      setEditingDoc(null)
      loadDocumentos()
    } catch (error) {
      console.error('Error actualizando precio:', error)
    }
  }

  const handleActualizarMensaje = async (id, nuevoMensaje) => {
    try {
      await actualizarMensaje(id, nuevoMensaje)
      setEditingDoc(null)
      loadDocumentos()
    } catch (error) {
      console.error('Error actualizando mensaje:', error)
    }
  }

  const handleReplaceClick = (docId) => {
    setReplacingPdf(docId)
    // Disparar el input file cuando se establece el id
    setTimeout(() => {
      if (fileInputRef.current) {
        fileInputRef.current.click()
      }
    }, 0)
  }

  const handleFileSelected = async (e) => {
    const file = e.target.files?.[0]
    if (!file || !replacingPdf) return

    try {
      // Buscar el documento actual para obtener sus datos
      const docActual = documentos.find(d => d.id === replacingPdf)
      if (!docActual) return

      // Primero eliminar el documento anterior
      await eliminarDocumento(replacingPdf)
      
      // Luego subir el nuevo documento con los mismos datos
      await subirDocumento(
        empresaId,
        file,
        docActual.campania_id || '',
        docActual.mensaje_entrega || '',
        docActual.precio || 0
      )
      
      // Limpiar y recargar
      setReplacingPdf(null)
      if (fileInputRef.current) fileInputRef.current.value = ''
      loadDocumentos()
      
    } catch (error) {
      console.error('Error reemplazando PDF:', error)
      alert('Error al reemplazar el PDF. Por favor intenta de nuevo.')
    }
  }

  if (loading) return <div className="campanas__loader">Cargando campañas...</div>

  return (
    <div className="campanas">
      <div className="campanas__header">
        <h1 className="campanas__title">Campañas</h1>
        <button className="campanas__btn-primary" onClick={() => setShowModal(true)}>
          <Upload size={18} />
          Nueva Campaña
        </button>
      </div>

      {/* Input oculto para reemplazar PDF */}
      <input
        type="file"
        ref={fileInputRef}
        onChange={handleFileSelected}
        accept=".pdf,.odf"
        style={{ display: 'none' }}
      />

      <div className="campanas__table-wrap">
        <table className="campanas__table">
          <thead>
            <tr>
              <th>ID</th>
              <th>Nombre</th>
              <th>Campaña</th>
              <th>Precio</th>
              <th>Mensaje de Entrega</th>
              <th>Fecha</th>
              <th>Acciones</th>
            </tr>
          </thead>
          <tbody>
            {documentos.map((doc) => (
              <tr key={doc.id} className="animate-fade-in">
                <td className="campanas__td-id">{doc.id}</td>
                <td className="campanas__td-name">{doc.nombre}</td>
                <td>
                  <span className="campanas__badge">{doc.campania_id || '—'}</span>
                </td>
                <td className="campanas__td-price">
                  {editingDoc?.id === doc.id && editingDoc?.tipo === 'precio' ? (
                    <div className="campanas__edit-price">
                      <input
                        type="number"
                        step="0.01"
                        defaultValue={doc.precio || 0}
                        id={`price-${doc.id}`}
                        className="campanas__input"
                      />
                      <button
                        className="campanas__btn-icon campanas__btn-save"
                        onClick={() => {
                          const input = document.getElementById(`price-${doc.id}`)
                          handleActualizarPrecio(doc.id, parseFloat(input.value))
                        }}
                      >
                        <Save size={14} />
                      </button>
                      <button
                        className="campanas__btn-icon campanas__btn-cancel"
                        onClick={() => setEditingDoc(null)}
                      >
                        <X size={14} />
                      </button>
                    </div>
                  ) : (
                    <>
                      ${doc.precio?.toFixed(2) || '0.00'}
                      <button
                        className="campanas__btn-icon campanas__btn-edit"
                        onClick={() => setEditingDoc({ id: doc.id, tipo: 'precio', valor: doc.precio || 0 })}
                        title="Editar precio"
                      >
                        <Edit2 size={14} />
                      </button>
                    </>
                  )}
                </td>
                <td className="campanas__td-message">
                  {editingDoc?.id === doc.id && editingDoc?.tipo === 'mensaje' ? (
                    <div className="campanas__edit-message">
                      <textarea
                        defaultValue={doc.mensaje_entrega || ''}
                        id={`msg-${doc.id}`}
                        className="campanas__textarea"
                        rows={2}
                      />
                      <div className="campanas__edit-actions">
                        <button
                          className="campanas__btn-icon campanas__btn-save"
                          onClick={() => {
                            const textarea = document.getElementById(`msg-${doc.id}`)
                            handleActualizarMensaje(doc.id, textarea.value)
                          }}
                        >
                          <Save size={14} />
                        </button>
                        <button
                          className="campanas__btn-icon campanas__btn-cancel"
                          onClick={() => setEditingDoc(null)}
                        >
                          <X size={14} />
                        </button>
                      </div>
                    </div>
                  ) : (
                    <>
                      {doc.mensaje_entrega ? doc.mensaje_entrega.substring(0, 30) + '...' : '—'}
                      <button
                        className="campanas__btn-icon campanas__btn-edit"
                        onClick={() => setEditingDoc({ id: doc.id, tipo: 'mensaje', valor: doc.mensaje_entrega || '' })}
                        title="Editar mensaje"
                      >
                        <MessageSquare size={14} />
                      </button>
                    </>
                  )}
                </td>
                <td>{new Date(doc.fecha_subida).toLocaleDateString()}</td>
                <td>
                  <div style={{ display: 'flex', gap: '8px' }}>
                    <button
                      className="campanas__btn-icon campanas__btn-replace"
                      onClick={() => handleReplaceClick(doc.id)}
                      title="Reemplazar PDF"
                    >
                      <File size={16} />
                    </button>
                    <button
                      className="campanas__btn-icon campanas__btn-delete"
                      onClick={() => handleEliminar(doc.id)}
                      title="Eliminar"
                    >
                      <Trash2 size={16} />
                    </button>
                  </div>
                </td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>

      {/* Modal para nueva campaña */}
      {showModal && (
        <div className="campanas__modal-overlay" onClick={() => setShowModal(false)}>
          <div className="campanas__modal" onClick={(e) => e.stopPropagation()}>
            <div className="campanas__modal-header">
              <h2>Nueva Campaña</h2>
              <button className="campanas__btn-close" onClick={() => setShowModal(false)}>×</button>
            </div>
            <div className="campanas__modal-body">
              <div className="campanas__form-group">
                <label>Archivo PDF</label>
                <input
                  type="file"
                  accept=".pdf,.odf"
                  onChange={(e) => setFormData({ ...formData, archivo: e.target.files?.[0] || null })}
                  className="campanas__file-input"
                />
              </div>
              <div className="campanas__form-group">
                <label>ID de Campaña</label>
                <input
                  type="text"
                  value={formData.campania_id}
                  onChange={(e) => setFormData({ ...formData, campania_id: e.target.value })}
                  placeholder="ej: lettering, reposteria"
                  className="campanas__input"
                />
              </div>
              <div className="campanas__form-group">
                <label>Precio</label>
                <input
                  type="number"
                  step="0.01"
                  value={formData.precio}
                  onChange={(e) => setFormData({ ...formData, precio: e.target.value })}
                  placeholder="0.00"
                  className="campanas__input"
                />
              </div>
              <div className="campanas__form-group">
                <label>Mensaje de Entrega</label>
                <textarea
                  value={formData.mensaje_entrega}
                  onChange={(e) => setFormData({ ...formData, mensaje_entrega: e.target.value })}
                  placeholder="Mensaje que se enviará al aprobar el pago..."
                  rows={4}
                  className="campanas__textarea"
                />
              </div>
            </div>
            <div className="campanas__modal-footer">
              <button className="campanas__btn-secondary" onClick={() => setShowModal(false)}>
                Cancelar
              </button>
              <button
                className="campanas__btn-primary"
                onClick={handleSubir}
                disabled={!formData.archivo || !formData.campania_id || !formData.mensaje_entrega || !formData.precio}
              >
                Subir Campaña
              </button>
            </div>
          </div>
        </div>
      )}
    </div>
  )
}
